name: Generate Video (Veo3 via pollo.ai) on: workflow_dispatch: inputs: prompt: description: "Video açıklaması (prompt)" required: true type: string negativePrompt: description: "Negatif prompt (opsiyonel)" required: false default: "" type: string image: description: "Referans görsel URL/base64 (opsiyonel)" required: false default: "" type: string length: description: "Süre (sn)" required: true default: "8" type: number aspectRatio: description: "En/Boy oranı" required: true default: "16:9" type: choice options: ["16:9","9:16","1:1","4:3","21:9"] resolution: description: "Çözünürlük" required: true default: "720p" type: choice options: ["480p","720p","1080p"] seed: description: "Seed (sayı, opsiyonel)" required: false default: "123" type: number generateAudio: description: "Ses üret (true/false)" required: true default: "true" type: choice options: ["true","false"] webhookUrl: description: "Webhook URL (opsiyonel)" required: false default: "" type: string jobs: make-video: runs-on: ubuntu-latest steps: - name: Checkout uses: actions/checkout@v4 - name: Install jq run: sudo apt-get update && sudo apt-get install -y jq - name: Build JSON payload run: | jq -n \ --arg prompt "${{ inputs.prompt }}" \ --arg negativePrompt "${{ inputs.negativePrompt }}" \ --arg image "${{ inputs.image }}" \ --arg aspect "${{ inputs.aspectRatio }}" \ --arg res "${{ inputs.resolution }}" \ --arg webhook "${{ inputs.webhookUrl }}" \ --argjson length ${{ inputs.length }} \ --argjson seed ${{ inputs.seed }} \ --argjson generateAudio ${{ inputs.generateAudio == 'true' }} \ ' { input: { prompt: $prompt, negativePrompt: $negativePrompt, image: $image, length: $length, aspectRatio: $aspect, resolution: $res, seed: $seed, generateAudio: $generateAudio }, webhookUrl: $webhook } ' > payload.json echo "Payload built:" cat payload.json - name: Call pollo.ai (Veo3) run: | curl --fail --request POST \ --url https://pollo.ai/api/platform/generation/google/veo3 \ --header "Content-Type: application/json" \ --header "x-api-key: ${{ secrets.POLLO_API_KEY }}" \ --data @payload.json \ | tee response.json - name: Save API response uses: actions/upload-artifact@v4 with: name: veo3-response path: response.json
